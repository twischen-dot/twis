    def calculate_angle(self, a, b, c):
        a, b, c = np.array(a), np.array(b), np.array(c)
        radians = np.arctan2(c[1]-b[1], c[0]-b[0]) - np.arctan2(a[1]-b[1], a[0]-b[0])
        angle = np.abs(radians * 180.0 / np.pi)
        return 360 - angle if angle > 180 else angle
    
    def draw_body_landmarks(self, frame, landmarks):
        h, w, _ = frame.shape
        for idx in range(11, 33):
            lm = landmarks[idx]
            if lm.visibility > 0.5:
                cv2.circle(frame, (int(lm.x * w), int(lm.y * h)), 5, (0, 255, 255), -1)
        
        for start_idx, end_idx in self.body_connections:
            s, e = landmarks[start_idx], landmarks[end_idx]
            if s.visibility > 0.5 and e.visibility > 0.5:
                cv2.line(frame, (int(s.x * w), int(s.y * h)), (int(e.x * w), int(e.y * h)), (0, 255, 0), 2)
    
    def process_frame(self, frame):
        rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        results = self.pose.process(rgb)
        
        if results.pose_landmarks:
            landmarks = results.pose_landmarks.landmark
            self.draw_body_landmarks(frame, landmarks)
            
            h, w, _ = frame.shape
            current_time = self.current_frame / self.fps
            
            # è®¡ç®—è§’åº¦
            def get_pt(idx): return [landmarks[idx].x * w, landmarks[idx].y * h]
            
            r_elbow_angle = self.calculate_angle(get_pt(12), get_pt(14), get_pt(16))
            l_elbow_angle = self.calculate_angle(get_pt(11), get_pt(13), get_pt(15))
            r_knee_angle = self.calculate_angle(get_pt(24), get_pt(26), get_pt(28))
            l_knee_angle = self.calculate_angle(get_pt(23), get_pt(25), get_pt(27))
            
            # æ£€æµ‹å‡ºæ‹³
            if (r_elbow_angle > 155 or l_elbow_angle > 155) and current_time - self.last_punch > 0.3:
                self.punch_count += 1
                self.last_punch = current_time
                cv2.putText(frame, "PUNCH!", (w//2-100, 80), cv2.FONT_HERSHEY_SIMPLEX, 2, (0, 255, 0), 3)
            
            # æ£€æµ‹è¸¢è…¿
            kick = (r_knee_angle > 150 and landmarks[28].y < landmarks[24].y) or \
                   (l_knee_angle > 150 and landmarks[27].y < landmarks[23].y)
            if kick and current_time - self.last_kick > 0.4:
                self.kick_count += 1
                self.last_kick = current_time
                cv2.putText(frame, "KICK!", (w//2-80, 140), cv2.FONT_HERSHEY_SIMPLEX, 2, (0, 128, 255), 3)
            
            # æ˜¾ç¤ºè§’åº¦
            for i, (name, val) in enumerate([("R Arm", r_elbow_angle), ("L Arm", l_elbow_angle), 
                                              ("R Leg", r_knee_angle), ("L Leg", l_knee_angle)]):
                cv2.putText(frame, f"{name}: {int(val)}", (10, 30+i*30), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255,255,255), 2)
        
        return frame
    
    def show_frame(self):
        self.cap.set(cv2.CAP_PROP_POS_FRAMES, self.current_frame)
        ret, frame = self.cap.read()
        
        if ret:
            frame = self.process_frame(frame)
            frame = cv2.resize(frame, (960, 540))
            frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            
            h, w, ch = frame.shape
            img = QImage(frame.data, w, h, ch * w, QImage.Format_RGB888)
            self.video_label.setPixmap(QPixmap.fromImage(img))
            
            self.punch_label.setText(f"å‡ºæ‹³: {self.punch_count}")
            self.kick_label.setText(f"è¸¢è…¿: {self.kick_count}")
            
            ct = self.current_frame / self.fps
            tt = self.total_frames / self.fps
            self.time_label.setText(f"{int(ct//60):02d}:{int(ct%60):02d} / {int(tt//60):02d}:{int(tt%60):02d}")
            self.progress.setValue(self.current_frame)
    
    def play_loop(self):
        if self.playing and self.current_frame < self.total_frames - 1:
            self.current_frame += 1
            self.show_frame()
        else:
            self.playing = False
            self.play_btn.setText("â–¶ æ’­æ”¾")
            self.audio_player.pause()
    
    def toggle_play(self):
        if self.playing:
            self.playing = False
            self.play_btn.setText("â–¶ æ’­æ”¾")
            self.timer.stop()
            self.audio_player.pause()
        else:
            self.playing = True
            self.play_btn.setText("â¸ æš‚åœ")
            self.audio_player.setPosition(int(self.current_frame / self.fps * 1000))
            self.audio_player.play()
            self.timer.start(int(1000 / self.fps))
    
    def stop(self):
        self.playing = False
        self.play_btn.setText("â–¶ æ’­æ”¾")
        self.timer.stop()
        self.audio_player.stop()
        self.current_frame = 0
        self.punch_count = self.kick_count = 0
        self.last_punch = self.last_kick = 0
        self.show_frame()
    
    def skip(self, seconds):
        self.current_frame = max(0, min(self.current_frame + int(seconds * self.fps), self.total_frames - 1))
        self.show_frame()
        if self.playing:
            self.audio_player.setPosition(int(self.current_frame / self.fps * 1000))
    
    def on_slider_pressed(self):
        self.timer.stop()
        
    def on_slider_released(self):
        self.current_frame = self.progress.value()
        self.show_frame()
        if self.playing:
            self.audio_player.setPosition(int(self.current_frame / self.fps * 1000))
            self.timer.start(int(1000 / self.fps))
    
    def on_volume_change(self, val):
        self.audio_player.setVolume(val)
    
    def closeEvent(self, event):
        self.timer.stop()
        self.audio_player.stop()
        self.cap.release()
